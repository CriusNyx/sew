# lines
    
  Returns: transformer
   
  Overloads:
    (predicate: predicate, transformer: transformer)
    (transformer: transformer)

  Flags:
    
  
  Description:
    For each line that matches the predicate apply the supplied transformer to it. If no predicate is supplied apply to every line instead.

# split
    
  Returns: transformer
   
  Overloads:
    (separator: string = "\n")

  Flags:
    
  
  Description:
    Split the string using the provided splitter

  Returns: transformer
   
  Overloads:

  Flags:
    
  
  Description:
    Join the string with the provided joiner.

# trim
    
  Returns: transformer
   
  Overloads:
    (trimCharacters: string = null)

  Flags:
    
  
  Description:
    Trim all characters provided in the input from the string. Trims whitespace by default.

# trimL, trimLeft, trimStart
    
  Returns: transformer
   
  Overloads:
    (trimCharacters: string = null)

  Flags:
    
  
  Description:
    Trim all characters provided from the start of the string. Trims whitespace by default.

# trimR, trimRight, trimEnd
    
  Returns: transformer
   
  Overloads:
    (trimCharacters: string = null)

  Flags:
    
  
  Description:
    Trim all characters provided from the end of the string. Trims whitespace by default.

# quote, quoted
    
  Returns: transformer
   
  Overloads:
    (quoteCharacter: string = "")

  Flags:
    
  
  Description:
    Surround the string in quotes.

# embed
    
  Returns: transformer
   
  Overloads:
    (start: string = null, end: string = null)

  Flags:
    
  
  Description:

# tag
    
  Returns: transformer
   
  Overloads:
    (tag: string = null)

  Flags:
    newline: boolean = false
  
  Description:
    Surround the string in HTML tags. If the newline flag is set.

  Returns: predicate
   
  Overloads:
